Loading data

data = 
  ny_noaa |> 
  drop_na() |> 
  separate(date, into = c("year", "month", "day"), "-") |> 
  mutate(
    tmax = as.numeric(tmax),
    tmin = as.numeric(tmin),
    prcp = as.numeric(prcp),
    snow = as.numeric(snow)
  ) |> 
  filter(year == 2000:2010) |> 
  group_by(year, month) |> 
  summarize(
    mean_prcp = mean(prcp),
    mean_snow = mean(snow),
    mean_tmax = mean(tmax),
    mean_tmin = mean(tmin)
  ) |> 
  mutate(
    month = case_when(
       month %in% c("01") ~ "Jan",
       month %in% c("02") ~ "Feb",
       month %in% c("03") ~ "Mar",
       month %in% c("04") ~ "Apr",
       month %in% c("05") ~ "May",
       month %in% c("06") ~ "Jun",
       month %in% c("07") ~ "Jul",
       month %in% c("08") ~ "Aug",
       month %in% c("09") ~ "Sep",
       month %in% c("10") ~ "Oct",
       month %in% c("11") ~ "Nov",
       month %in% c("12") ~ "Dec"),
    month = factor(month, levels = c(
        "Jan", "Feb", "Mar", "Apr", "May", "Jun",
        "Jul", "Aug", "Sep", "Oct", "Nov", "Dec")))
